  <div class="box"><h3 style="margin-top:0">Letzte Verteilung</h3><div id="pbt-weights"></div></div>
  <div class="box"><h3 style="margin-top:0">V6 Diagnose</h3><div id="pbt-diagnostics"></div></div>
  <div class="box"><h3 style="margin-top:0">Meta</h3><div id="pbt-meta"></div></div>

  <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>
  <script>
    (function(){
      var API_URL="https://portfolio-webapp-kkxd.onrender.com/api/backtest_v6";
      var btn=document.getElementById("pbt-run");
      var statusEl=document.getElementById("pbt-status");
      var rawEl=document.getElementById("pbt-raw");
      var summaryEl=document.getElementById("pbt-summary");
      var weightsEl=document.getElementById("pbt-weights");
      var diagEl=document.getElementById("pbt-diagnostics");
      var metaEl=document.getElementById("pbt-meta");
      var chartCanvas=document.getElementById("pbt-chart");
      var chartInstance=null;

      function esc(x){return String(x==null?"":x).replace(/[&<>"]/g,function(c){return {"&":"&amp;","<":"&lt;",">":"&gt;","\"":"&quot;"}[c];});}
      function pct(x,d){return ((Number(x)||0)*100).toFixed(d==null?2:d)+" %";}

      function renderSummary(summary){
        if(!summary||!summary.length){summaryEl.innerHTML="<p>Keine Daten</p>";return;}
        var html='<table><thead><tr><th>Strategie</th><th>CAGR</th><th>Vola</th><th>Max DD</th><th>Sharpe</th></tr></thead><tbody>';
        summary.forEach(function(row){html+='<tr><td>'+esc(row.strategy)+'</td><td>'+esc(row.cagr)+'</td><td>'+esc(row.vola)+'</td><td>'+esc(row.max_dd)+'</td><td>'+esc(row.sharpe)+'</td></tr>';});
        html+='</tbody></table>';summaryEl.innerHTML=html;
      }

      function renderWeights(latest){
        if(!latest||!latest.weights){weightsEl.innerHTML="<p>Keine Daten</p>";return;}
        var keys=Object.keys(latest.weights).sort(function(a,b){return latest.weights[b]-latest.weights[a];});
        var html='<p><strong>Datum:</strong> '+esc(latest.date||'-')+'</p>';
        html+='<table><thead><tr><th>Asset</th><th>Gewicht</th></tr></thead><tbody>';
        keys.forEach(function(k){html+='<tr><td>'+esc(k)+'</td><td>'+pct(latest.weights[k],2)+'</td></tr>';});
        html+='</tbody></table>';
        html+='<table><thead><tr><th>Kennzahl</th><th>Wert</th></tr></thead><tbody>';
        html+='<tr><td>Turnover</td><td>'+pct(latest.turnover,2)+'</td></tr>';
        html+='<tr><td>Rebalancing-Kosten</td><td>'+pct(latest.rebalancing_cost,3)+'</td></tr>';
        html+='<tr><td>Bruttorendite nächster Monat</td><td>'+pct(latest.gross_return,2)+'</td></tr>';
        html+='<tr><td>Nettorendite nächster Monat</td><td>'+pct(latest.net_return,2)+'</td></tr>';
        html+='</tbody></table>';weightsEl.innerHTML=html;
      }

      function renderDiagnostics(latest){
        if(!latest||!latest.v6){diagEl.innerHTML="<p>Keine Diagnose verfügbar</p>";return;}
        var v6=latest.v6;
        var ce=v6.commodity_engine||{};
        var ee=v6.equity_engine||{};
        var html='<table><tbody>';
        html+='<tr><th>Ausgewählte Asset-Level-Bausteine</th><td>'+esc((v6.selected_assets||[]).join(', ')||'-')+'</td></tr>';
        html+='<tr><th>Asset-Level-Gewichte</th><td><pre>'+esc(JSON.stringify(v6.asset_level_weights||{},null,2))+'</pre></td></tr>';
        html+='<tr><th>Commodity Engine interne Gewichte</th><td><pre>'+esc(JSON.stringify(ce.internal_weights||{},null,2))+'</pre></td></tr>';
        html+='<tr><th>Commodity Scores</th><td><pre>'+esc(JSON.stringify(ce.bucket_scores||{},null,2))+'</pre></td></tr>';
        html+='<tr><th>Commodity Auswahl</th><td>'+esc((ce.selected_buckets||[]).join(', ')||'-')+'</td></tr>';
        html+='<tr><th>Equity Engine interne Gewichte</th><td><pre>'+esc(JSON.stringify(ee.internal_weights||{},null,2))+'</pre></td></tr>';
        html+='<tr><th>Fallback</th><td>'+esc(v6.fallback_used?'ja':'nein')+(v6.fallback_reason?' — '+esc(v6.fallback_reason):'')+'</td></tr>';
        html+='</tbody></table>';diagEl.innerHTML=html;
      }

      function renderMeta(meta){
        if(!meta){metaEl.innerHTML="<p>Keine Daten</p>";return;}
        var html='<table><tbody>';
        html+='<tr><th>Version</th><td>'+esc(meta.version||'v6')+'</td></tr>';
        html+='<tr><th>Bitcoin</th><td>'+esc(meta.include_bitcoin?'an':'aus')+'</td></tr>';
        html+='<tr><th>Rebalancing-Kosten</th><td>'+pct(meta.transaction_cost_rate,2)+'</td></tr>';
        html+='<tr><th>Equity-Sektor Max</th><td>'+pct(meta.equity_factor_max,1)+'</td></tr>';
        html+='<tr><th>Commodity Engine</th><td>'+esc(meta.commodity_engine_weighting||'-')+'</td></tr>';
        html+='<tr><th>Commodity Gesamt-Cap</th><td>'+pct(meta.max_total_commodities,1)+'</td></tr>';
        html+='<tr><th>Commodity Broad-Cap</th><td>'+pct(meta.max_broad_commodities,1)+'</td></tr>';
        html+='<tr><th>Commodity Sub-Cap</th><td>'+pct(meta.max_single_commodity_sub,1)+'</td></tr>';
        html+='<tr><th>Gold</th><td>'+esc(meta.gold_treatment||'separat')+'</td></tr>';
        html+='<tr><th>Start effektiv</th><td>'+esc(meta.actual_start_date||'-')+'</td></tr>';
        if(meta.proxies){html+='<tr><th>Proxies</th><td><pre>'+esc(JSON.stringify(meta.proxies,null,2))+'</pre></td></tr>';}
        html+='</tbody></table>';metaEl.innerHTML=html;
      }

      function renderChart(chart){
        if(!chart||!chart.dates||!chart.portfolio||!chart.benchmark){return;}
        if(chartInstance){chartInstance.destroy();}
        chartInstance=new Chart(chartCanvas.getContext("2d"),{
          type:"line",
          data:{labels:chart.dates,datasets:[
            {label:"Portfolio V6",data:chart.portfolio,borderWidth:2,pointRadius:0,tension:0.15},
            {label:"Benchmark 70/30",data:chart.benchmark,borderWidth:2,pointRadius:0,tension:0.15}
          ]},
          options:{responsive:true,maintainAspectRatio:false,interaction:{mode:"index",intersect:false},plugins:{legend:{display:true},tooltip:{callbacks:{label:function(context){var val=context.parsed.y;return context.dataset.label+": "+(typeof val==="number"?val.toFixed(3):val);}}}},scales:{x:{ticks:{maxTicksLimit:12}},y:{beginAtZero:false}}}
        });
      }

      btn.addEventListener("click",function(){
        var cost=parseFloat(String(document.getElementById("pbt-cost").value).replace(",","."));
        var equityFactorMax=parseFloat(String(document.getElementById("pbt-equity-factor-max").value).replace(",","."));
        var startDate=String(document.getElementById("pbt-start").value||"").trim();
        var includeBitcoin=document.getElementById("pbt-bitcoin").checked;
        var sectorProxySet=document.getElementById("pbt-sector-set").value;
        if(!isFinite(cost)){statusEl.className="error";statusEl.textContent="Ungültiger Kostenwert";return;}
        if(!isFinite(equityFactorMax)){statusEl.className="error";statusEl.textContent="Ungültiger Equity-Factor-Wert";return;}
        statusEl.className="muted";statusEl.textContent="Lädt ...";
        fetch(API_URL,{method:"POST",headers:{"Content-Type":"application/json","Accept":"application/json"},body:JSON.stringify({transaction_cost_rate:cost,start_date:startDate||null,equity_factor_max:equityFactorMax,include_bitcoin:includeBitcoin,sector_proxy_set:sectorProxySet})})
        .then(function(res){return res.json().then(function(data){return {ok:res.ok,data:data};});})
        .then(function(result){
          if(!result.ok){throw new Error(result.data&&result.data.error?result.data.error:"API-Fehler");}
          statusEl.className="success";statusEl.textContent="Backtest V6 erfolgreich geladen";
          renderChart(result.data.chart);
          renderSummary(result.data.summary);
          renderWeights(result.data.latest_weights);
          renderDiagnostics(result.data.latest_weights);
          renderMeta(result.data.meta);
          rawEl.textContent=JSON.stringify(result.data,null,2);
        })
        .catch(function(err){statusEl.className="error";statusEl.textContent="Fehler: "+err.message;rawEl.textContent=String(err&&err.stack?err.stack:err);});
      });
    })();
  </script>
</div>
