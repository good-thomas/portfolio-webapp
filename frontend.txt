  <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>
  <script>
    (function () {
      var API_URL = "https://portfolio-webapp-kkxd.onrender.com/api/backtest";

      var btn = document.getElementById("pbt-run");
      var statusEl = document.getElementById("pbt-status");
      var rawEl = document.getElementById("pbt-raw");
      var summaryEl = document.getElementById("pbt-summary");
      var weightsEl = document.getElementById("pbt-weights");
      var metaEl = document.getElementById("pbt-meta");
      var chartCanvas = document.getElementById("pbt-chart");

      var chartInstance = null;

      function renderSummary(summary) {
        if (!summary || !summary.length) {
          summaryEl.innerHTML = "<p>Keine Daten</p>";
          return;
        }

        var html = '<table><thead><tr><th>Strategie</th><th>CAGR</th><th>Vola</th><th>Max DD</th><th>Sharpe</th></tr></thead><tbody>';

        summary.forEach(function (row) {
          html += '<tr>' +
            '<td>' + (row.strategy || '') + '</td>' +
            '<td>' + (row.cagr || '') + '</td>' +
            '<td>' + (row.vola || '') + '</td>' +
            '<td>' + (row.max_dd || '') + '</td>' +
            '<td>' + (row.sharpe || '') + '</td>' +
            '</tr>';
        });

        html += '</tbody></table>';
        summaryEl.innerHTML = html;
      }

      function renderWeights(latest) {
        if (!latest || !latest.weights) {
          weightsEl.innerHTML = "<p>Keine Daten</p>";
          return;
        }

        var html = '';
        html += '<p><strong>Datum:</strong> ' + (latest.date || '-') + '</p>';
        html += '<p><strong>Aktive Assets:</strong> ' + ((latest.active_assets || []).join(', ') || '-') + '</p>';

        html += '<table><thead><tr><th>Asset</th><th>Gewicht</th></tr></thead><tbody>';

        Object.keys(latest.weights).forEach(function (key) {
          var val = latest.weights[key];
          html += '<tr><td>' + key + '</td><td>' + (val * 100).toFixed(2) + ' %</td></tr>';
        });

        html += '</tbody></table>';

        html += '<table><thead><tr><th>Kennzahl</th><th>Wert</th></tr></thead><tbody>';
        html += '<tr><td>Turnover</td><td>' + ((latest.turnover || 0) * 100).toFixed(2) + ' %</td></tr>';
        html += '<tr><td>Rebalancing-Kosten</td><td>' + ((latest.rebalancing_cost || 0) * 100).toFixed(3) + ' %</td></tr>';
        html += '<tr><td>Bruttorendite</td><td>' + ((latest.gross_return || 0) * 100).toFixed(2) + ' %</td></tr>';
        html += '<tr><td>Nettorendite</td><td>' + ((latest.net_return || 0) * 100).toFixed(2) + ' %</td></tr>';
        html += '</tbody></table>';

        weightsEl.innerHTML = html;
      }

      function renderMeta(meta) {
        if (!meta) {
          metaEl.innerHTML = "<p>Keine Daten</p>";
          return;
        }

        var html = '<table><tbody>';
        html += '<tr><th>Managed Futures</th><td>' + (meta.use_managed_futures ? 'an' : 'aus') + '</td></tr>';
        html += '<tr><th>Rebalancing-Kosten</th><td>' + ((meta.transaction_cost_rate || 0) * 100).toFixed(2) + ' %</td></tr>';
        html += '<tr><th>Benchmark</th><td>70 % ACWI / 30 % IEF</td></tr>';

        if (meta.proxies) {
          html += '<tr><th>Proxies</th><td><pre>' + JSON.stringify(meta.proxies, null, 2) + '</pre></td></tr>';
        }

        html += '</tbody></table>';
        metaEl.innerHTML = html;
      }

      function renderChart(chart) {
        if (!chart || !chart.dates || !chart.portfolio || !chart.benchmark) {
          return;
        }

        if (chartInstance) {
          chartInstance.destroy();
        }

        chartInstance = new Chart(chartCanvas.getContext("2d"), {
          type: "line",
          data: {
            labels: chart.dates,
            datasets: [
              {
                label: "Portfolio",
                data: chart.portfolio,
                borderWidth: 2,
                pointRadius: 0,
                tension: 0.15
              },
              {
                label: "Benchmark 70/30",
                data: chart.benchmark,
                borderWidth: 2,
                pointRadius: 0,
                tension: 0.15
              }
            ]
          },
          options: {
            responsive: true,
            maintainAspectRatio: false,
            interaction: {
              mode: "index",
              intersect: false
            },
            plugins: {
              legend: {
                display: true
              },
              tooltip: {
                callbacks: {
                  label: function(context) {
                    var val = context.parsed.y;
                    return context.dataset.label + ": " + (typeof val === "number" ? val.toFixed(3) : val);
                  }
                }
              }
            },
            scales: {
              x: {
                ticks: {
                  maxTicksLimit: 12
                }
              },
              y: {
                beginAtZero: false
              }
            }
          }
        });
      }

      btn.addEventListener("click", function () {
        var cost = parseFloat(String(document.getElementById("pbt-cost").value).replace(",", "."));
        var useMF = document.getElementById("pbt-mf").checked;

        if (!isFinite(cost)) {
          statusEl.className = "error";
          statusEl.textContent = "Ungültiger Kostenwert";
          return;
        }

        statusEl.className = "muted";
        statusEl.textContent = "Lädt ...";

        fetch(API_URL, {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            "Accept": "application/json"
          },
          body: JSON.stringify({
            transaction_cost_rate: cost,
            use_managed_futures: useMF
          })
        })
        .then(function (res) {
          return res.json().then(function (data) {
            return { ok: res.ok, data: data };
          });
        })
        .then(function (result) {
          if (!result.ok) {
            throw new Error(result.data && result.data.error ? result.data.error : "API-Fehler");
          }

          statusEl.className = "success";
          statusEl.textContent = "Backtest erfolgreich geladen";

          renderChart(result.data.chart);
          renderSummary(result.data.summary);
          renderWeights(result.data.latest_weights);
          renderMeta(result.data.meta);
          rawEl.textContent = JSON.stringify(result.data, null, 2);
        })
        .catch(function (err) {
          statusEl.className = "error";
          statusEl.textContent = "Fehler: " + err.message;
          rawEl.textContent = String(err && err.stack ? err.stack : err);
        });
      });
    })();
  </script>
